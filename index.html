// Initialize Firebase firebase.initializeApp(firebaseConfig); const database = firebase.database(); // Get DOM elements const toggleBtn = document.getElementById('toggleBtn'); const statusDiv = document.getElementById('status'); // Reference to LED status in Firebase const ledRef = database.ref('esp32/led'); // Listen for LED state changes ledRef.on('value', (snapshot) => { const state = snapshot.val(); statusDiv.textContent = `LED: ${state ? 'ON' : 'OFF'}`; statusDiv.style.color = state ? 'green' : 'red'; }); // Toggle button click handler toggleBtn.addEventListener('click', () => { ledRef.once('value').then((snapshot) => { const currentState = snapshot.val(); ledRef.set(!currentState); }); });
